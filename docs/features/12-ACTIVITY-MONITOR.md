export const useActivitySimulation = () => {
  const [cpuHistory, setCpuHistory] = useState<number[]>([]);
  const [memoryHistory, setMemoryHistory] = useState<number[]>([]);
  
  useEffect(() => {
    const interval = setInterval(() => {
      // Simulate CPU fluctuation
      const newCpu = Math.random() * 30 + 20; // 20-50%
      setCpuHistory(prev => [...prev.slice(-59), newCpu]);
      
      // Simulate Memory (more stable)
      const newMemory = 40 + Math.random() * 5; // 40-45%
      setMemoryHistory(prev => [...prev.slice(-59), newMemory]);
    }, 1000);
    
    return () => clearInterval(interval);
  }, []);
  
  return { cpuHistory, memoryHistory };
};
