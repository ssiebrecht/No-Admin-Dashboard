// types/settings.ts
interface Settings {
  appearance: {
    theme: 'platinum' | 'graphite';
    highlightColor: string;
    fontSize: 'small' | 'medium' | 'large';
  };
  sound: {
    outputVolume: number;
    muted: boolean;
    alertSound: string;
    alertVolume: number;
  };
  dateTime: {
    use24Hour: boolean;
    showDateInMenuBar: boolean;
    timezone: string;
  };
  display: {
    scaling: number;
    brightness: number;
    autoAdjustBrightness: boolean;
  };
  keyboard: {
    keyRepeatSpeed: number;
    delayUntilRepeat: number;
  };
  mouse: {
    trackingSpeed: number;
    doubleClickSpeed: number;
    scrollDirection: 'natural' | 'classic';
  };
}
